相同尺寸的`Mat`对象进行位运算运算，即每个像素点的每个通道间进行位运算。

```cpp
int main()
{
	cv::Mat m1 = cv::Mat(cv::Size(255, 255), CV_8UC3);
	cv::Mat m2 = cv::Mat(cv::Size(255, 255), CV_8UC3);

	cv::rectangle(m1, cv::Rect(100, 100, 100, 100), cv::Scalar(255, 255, 0), -1);
	cv::rectangle(m2, cv::Rect(120, 120, 100, 100), cv::Scalar(0, 255, 255), -1);

	cv::imshow("m1", m1);
	cv::imshow("m2", m2);
	cv::imshow("m1 | m2", m1 | m2);
	cv::imshow("m1 & m2", m1 & m2);
	cv::imshow("m1 ^ m2", m1 ^ m2);
	cv::imshow("!m1", ~m1);

	cv::waitKey();

	return 0;
}
```
