#### 实现

```cpp
void selectSort(std::vector<int> &nums)
{
    if (nums.size() <= 1)
        return;

    for (int left = 0, right = nums.size() - 1; left < right; ++left, --right)
    {
        // 每次将最小的移动到左边，最大的移动到右边
        int min_index = left, max_index = right;
        for (int j = left; j <= right; ++j)
        {
            min_index = nums[j] < nums[min_index] ? j : min_index;
            max_index = nums[j] > nums[max_index] ? j : max_index;
        }
        // 当最大最小分别位于两端时，交换一次即可
        std::swap(nums[left], nums[min_index]);
        if (left != max_index)
            std::swap(nums[right], nums[max_index]);
    }
}
```
