#### 实现

```cpp
void bubbleSort(std::vector<int> &nums)
{
    if (nums.size() <= 1)
        return;

    for (int i = 0; i < nums.size() - 1; ++i)
    {
        // 如果某次冒泡，没有任何元素移动，则可以直接结束冒泡
        bool unbubbled = true;
        // 每次冒泡后，界限内最大元素都会被移动到最右边，所以界限内最右边的元素可以不用冒泡
        for (int j = 0; j < nums.size() - i - 1; ++j)
        {
            if (nums[j] > nums[j + 1])
            {
                std::swap(nums[j], nums[j + 1]);
                unbubbled = false;
            }
        }
        if (unbubbled)
            return;
    }
}
```
